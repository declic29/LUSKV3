document.getElementById('translate-btn').addEventListener('click', async function() { const sourceText = document.getElementById('source-text').value; const sourceLang = document.getElementById('source-lang').value; const targetLang = document.getElementById('target-lang').value; if (!sourceText.trim()) return; document.getElementById('target-text').value = "Traduction en cours..."; try { const response = await fetch('http://localhost:5000/translate', { method: 'POST', headers: { 'Content-Type': 'application/json', // Supprimez l'authentification si vous n'utilisez pas de clé // 'Authorization': 'Bearer votre_cle_secrete' }, body: JSON.stringify({ q: sourceText, source: sourceLang, target: targetLang }) }); if (!response.ok) throw new Error('Erreur serveur'); const data = await response.json(); document.getElementById('target-text').value = data.translatedText || "Erreur de traduction"; } catch (error) { console.error("Erreur détaillée:", error); document.getElementById('target-text').value = `Erreur : ${error.message}`; } });
